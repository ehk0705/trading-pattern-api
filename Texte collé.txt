function ouvrirTradingViewAvecIndicateur() {
    const selectActif = document.getElementById("asset-selector");
    const selectIndicateur = document.getElementById("indicator-selector");

    if (!selectActif) {
        alert("Sélecteur d’actif introuvable.");
        return;
    }

    if (!selectIndicateur) {
        alert("Sélecteur d’indicateur introuvable.");
        return;
    }

    const symbole = selectActif.value;
    const indicateur = selectIndicateur.value;

    if (!symbole) {
        alert("Aucun actif sélectionné.");
        return;
    }

    if (!indicateur) {
        alert("Aucun indicateur sélectionné.");
        return;
    }

    const scriptPine = genererScriptPine(indicateur);
    const urlTradingView = "https://www.tradingview.com/chart/?symbol=" + encodeURIComponent(symbole);

    if (scriptPine) {
        navigator.clipboard.writeText(scriptPine)
            .then(function () {
                alert(
                    "Actif sélectionné : " + symbole + "\n" +
                    "Indicateur sélectionné : " + indicateur + "\n\n" +
                    "Le script Pine Script a été copié.\n\n" +
                    "TradingView va maintenant s’ouvrir.\n" +
                    "Collez ensuite le script dans Pine Editor."
                );

                window.open(urlTradingView, "_blank");
            })
            .catch(function () {
                alert(
                    "TradingView va s’ouvrir avec l’actif sélectionné : " + symbole + ".\n\n" +
                    "Impossible de copier automatiquement le script Pine Script.\n" +
                    "Copiez-le manuellement depuis votre page de stratégie."
                );

                window.open(urlTradingView, "_blank");
            });
    } else {
        alert(
            "TradingView va s’ouvrir avec l’actif sélectionné : " + symbole + ".\n\n" +
            "Indicateur sélectionné : " + indicateur + "\n\n" +
            "Cet indicateur est personnalisé. TradingView ne permet pas de le charger automatiquement par URL.\n" +
            "Il faut l’ajouter manuellement dans TradingView ou le coller dans Pine Editor."
        );

        window.open(urlTradingView, "_blank");
    }
}

function genererScriptPine(indicateur) {
    if (indicateur === "RSI") {
        return `//@version=5
indicator("RSI pédagogique", overlay=false)

rsi = ta.rsi(close, 14)

plot(rsi, title="RSI", color=color.blue)
hline(70, "Zone de surachat")
hline(30, "Zone de survente")
hline(50, "Zone neutre")`;
    }

    if (indicateur === "MACD") {
        return `//@version=5
indicator("MACD pédagogique", overlay=false)

[macdLine, signalLine, histLine] = ta.macd(close, 12, 26, 9)

plot(macdLine, title="MACD", color=color.blue)
plot(signalLine, title="Signal", color=color.orange)
plot(histLine, title="Histogramme", style=plot.style_columns)`;
    }

    if (indicateur === "EMA") {
        return `//@version=5
indicator("EMA 20 / EMA 50 pédagogique", overlay=true)

ema20 = ta.ema(close, 20)
ema50 = ta.ema(close, 50)

plot(ema20, title="EMA 20", color=color.green)
plot(ema50, title="EMA 50", color=color.red)`;
    }

    if (indicateur === "SMA") {
        return `//@version=5
indicator("SMA 50 / SMA 200 pédagogique", overlay=true)

sma50 = ta.sma(close, 50)
sma200 = ta.sma(close, 200)

plot(sma50, title="SMA 50", color=color.blue)
plot(sma200, title="SMA 200", color=color.orange)`;
    }

    if (indicateur === "SUPERTREND") {
        return `//@version=5
indicator("Supertrend pédagogique", overlay=true)

facteur = input.float(3.0, title="Facteur")
periodeAtr = input.int(10, title="Période ATR")

[supertrend, direction] = ta.supertrend(facteur, periodeAtr)

plot(supertrend, title="Supertrend", color=direction < 0 ? color.green : color.red)`;
    }

    if (indicateur === "BOLLINGER") {
        return `//@version=5
indicator("Bandes de Bollinger pédagogiques", overlay=true)

periode = input.int(20, title="Période")
multiplicateur = input.float(2.0, title="Multiplicateur")

base = ta.sma(close, periode)
ecart = multiplicateur * ta.stdev(close, periode)

bandeHaute = base + ecart
bandeBasse = base - ecart

plot(base, title="Moyenne", color=color.blue)
plot(bandeHaute, title="Bande haute", color=color.orange)
plot(bandeBasse, title="Bande basse", color=color.orange)`;
    }

    return "";
}
